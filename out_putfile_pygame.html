import pygame from pyscript import document # Initialize pygame pygame.init() WIDTH, HEIGHT = 420, 500 # Get the container and create canvas container = document.getElementById("pygame_container") canvas = pygame.display.set_mode((WIDTH, HEIGHT)) canvas.fill((240, 240, 240)) # Widget rendering positions y_pos = 30 # StringVar storage stringvars = {} # Store button info for click handling buttons = {} # Render each widget using Pygame pygame.display.flip() # Main event loop running = True clock = pygame.time.Clock() while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.MOUSEBUTTONDOWN: # Check button clicks for btn_id, rect in buttons.items(): if rect.collidepoint(event.pos): print(f"Button {btn_id} clicked") # You can map to your Python functions here # if btn_id == "submit": submit_function() clock.tick(60) pygame.quit()
